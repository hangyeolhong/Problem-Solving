### Python solution
```python
from collections import deque

N, M = map(int, input().split())
graph = []

# -1: 검은색, 0:무지개, 1 ~ M: 일반
for _ in range(N):
    graph.append(list(map(int, input().split())))

answer = 0

dx, dy = [1, -1, 0, 0], [0, 0, 1, -1]


def bfs(x, y, visit, color):
    q = deque()
    q.append((x, y))
    res = []
    rainbow = 0  # 무지개 블록 수
    rainbow_pos = []
    visit.add((x, y))
    standard = (N, N)  # 기준 블록

    while q:
        a, b = q.popleft()
        if graph[a][b] == 0:
            rainbow += 1
            rainbow_pos.append((a, b))
        else:
            if (a, b) < standard:
                standard = (a, b)
        res.append((a, b))

        for i in range(4):
            na, nb = a + dx[i], b + dy[i]
            if na < 0 or na >= N or nb < 0 or nb >= N or (na, nb) in visit:
                continue

            # 무지개 블록이거나, 같은 색의 일반 블록인 경우로만 bfs
            if graph[na][nb] == 0 or graph[na][nb] == color:
                visit.add((na, nb))
                q.append((na, nb))

    # 무지개 블록은 방문 해제
    for a, b in rainbow_pos:
        visit.remove((a, b))

    return res, rainbow, standard


def gravity(g):
    for i in range(N - 2, -1, -1):
        for j in range(N):
            if g[i][j] > -1:
                # 검은색 블록이 아니면 밑으로 내려감
                r = i
                while True:
                    if 0 <= r + 1 < N and g[r + 1][j] == -10:
                        g[r + 1][j] = g[r][j]
                        g[r][j] = -10
                        r += 1
                    else:
                        break


def rotate(g):
    res = [[0] * N for _ in range(N)]
    for i in range(N):
        for j in range(N):
            res[N - 1 - j][i] = g[i][j]
    return res


t = 0

while True:
    if t == 1:
        break
    # step1
    visit = set()
    block_group = []

    for i in range(N):
        for j in range(N):
            if 1 <= graph[i][j] <= M and (i, j) not in visit:
                tmp_block, num_rainbow, standard_block = bfs(i, j, visit, graph[i][j])
                # 블록 크기, 무지개 블록의 수, 기준 블록의 행, 기준 블록의 열, 블록 좌표 리스트
                if len(tmp_block) >= 2:
                    block_group.append((len(tmp_block), num_rainbow, standard_block[0], standard_block[1], tmp_block))

    block_group.sort(key=lambda x: (-x[0], -x[1], -x[2], -x[3]))
    if not block_group:
        break

    # step2

    B, _, _, _, block_pos = block_group[0]
    for x, y in block_pos:
        graph[x][y] = -10  # 블록 제거
    answer += B ** 2

    # step3 : 중력 작용
    gravity(graph)

    # step4: 90도 반시계 회전
    graph = rotate(graph)

    # step5: 중력 작용
    gravity(graph)
    # for i in range(N):
    #     print(*graph[i])

print(answer)

```
