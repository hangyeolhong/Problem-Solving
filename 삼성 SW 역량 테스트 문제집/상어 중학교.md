```python
import sys
from collections import deque

N, M = map(int, sys.stdin.readline().rstrip().split())
graph = []

for _ in range(N):
    graph.append(list(map(int, sys.stdin.readline().rstrip().split())))

dx, dy = [1, -1, 0, 0], [0, 0, 1, -1]


def find_block(color, x, y, visit):
    q = deque()
    visit.add((x, y))
    rainbow = []
    q.append((x, y))
    block = []

    while q:
        i, j = q.popleft()
        block.append((i, j))
        for d in range(4):
            ni, nj = i + dx[d], j + dy[d]
            if ni < 0 or ni >= N or nj < 0 or nj >= N or (ni, nj) in visit:
                continue
            if graph[ni][nj] == color:
                q.append((ni, nj))
                visit.add((ni, nj))
            elif graph[ni][nj] == 0:
                q.append((ni, nj))
                visit.add((ni, nj))
                rainbow.append((ni, nj))

    # ë¬´ì§€ê°œ ë¸”ë¡ì€ ë‹¤ì‹œ íƒìƒ‰í•  ìˆ˜ ìˆë„ë¡ ë°©ë¬¸ ì²˜ë¦¬ë¥¼ ì·¨ì†Œí•´ì¤Œ
    for i, j in rainbow:
        visit.remove(((i, j)))

    return len(block), len(rainbow), (x, y), block


def gravity(g):
    for i in range(N - 2, -1, -1):
        for j in range(N):
            if g[i][j] > -1:
                r = i
                while True:
                    if 0 <= r + 1 < N and g[r + 1][j] == -2:
                        g[r + 1][j] = g[r][j]
                        g[r][j] = -2
                        r += 1
                    else:
                        break

    return g


def rotate(g):
    res = [[0] * N for _ in range(N)]

    for i in range(N):
        for j in range(N):
            res[N - 1 - j][i] = g[i][j]

    return res


score = 0
while True:
    # step 1
    visit = set()
    mx = []

    for i in range(N):
        for j in range(N):
            if graph[i][j] > 0 and (i, j) not in visit:
                block_size, rainbow_size, std_pos, block_pos = find_block(graph[i][j], i, j, visit)
                if block_size >= 2:
                    mx.append((block_size, rainbow_size, std_pos, block_pos))

    if not mx:
        break
    
    # step 2
    mx.sort(key=lambda x: (-x[0], -x[1], -x[2][0], -x[2][1]))
    B, _, _, block_pos = mx[0]
    score += B ** 2

    for x, y in block_pos:
        graph[x][y] = -2  # ë¹ˆì¹¸

    # step 3
    graph = gravity(graph)
    # step 4
    graph = rotate(graph)
    # step 5
    graph = gravity(graph)

print(score)
```
