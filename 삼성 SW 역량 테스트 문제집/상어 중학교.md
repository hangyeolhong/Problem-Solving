def bfs(x, y, visit, color):

                standard = (a, b)
        res.append((a, b))

        for i in range(4):

    # ë¬´ì§€ê°œ ë¸”ë¡ì€ ë°©ë¬¸ í•´ì œ
    for a, b in rainbow_pos:
        visit.remove((a, b))

    return res, rainbow, standard


def gravity(g):
    for i in range(N - 2, -1, -1):
        for j in range(N):
            if g[i][j] > -1:
                # ê²€ì€ìƒ‰ ë¸”ë¡ì´ ì•„ë‹ˆë©´ ë°‘ìœ¼ë¡œ ë‚´ë ¤ê°

                        g[r][j] = -10
                        r += 1
                    else:
                        break


def rotate(g):
    res = [[0] * N for _ in range(N)]
    for i in range(N):
        for j in range(N):
            res[N - 1 - j][i] = g[i][j]
    return res

                if len(tmp_block) >= 2:
