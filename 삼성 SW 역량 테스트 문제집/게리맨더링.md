### Python solution
```python
import sys
from itertools import combinations
from collections import deque

N = int(sys.stdin.readline().rstrip())
people = list(map(int, sys.stdin.readline().rstrip().split()))
graph = [[] for _ in range(N)]

for i in range(N):
    x = list(map(int, sys.stdin.readline().rstrip().split()))
    for j in range(x[0]):
        graph[i].append(x[j + 1] - 1)


def bfs(l):
    q = deque()
    q.append(l[0])
    visit.add(l[0])
    total = 0
    res = []

    while q:
        x = q.popleft()
        res.append(x)
        total += people[x]
        for neighbor in graph[x]:
            if neighbor not in visit and neighbor in l:
                q.append(neighbor)
                visit.add(neighbor)

    return res, total


answer = 1e9

for size in range(1, N // 2 + 1):
    for case in combinations(range(N), size):
        visit = set()
        a, s1 = bfs(case)
        b, s2 = bfs([i for i in range(N) if i not in case])

        if len(a) + len(b) == N:
            answer = min(answer, abs(s1 - s2))

print(-1 if answer == 1e9 else answer)
```
